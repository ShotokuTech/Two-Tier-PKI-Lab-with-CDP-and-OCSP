function Set-DriveLabel($letter='C:', $label='ISOC-GDC0_C') {
if (!(Test-Path $letter)) {
Throw "Drive $letter does not exist."
}
$instance = ([wmi]"Win32_LogicalDisk='$letter'")
$instance.VolumeName = $label
$instance.Put()
}
